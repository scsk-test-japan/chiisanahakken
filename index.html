/** * Gemini プロキシ（GAS Web App） * - APIキーは Script Properties に保管し、コードにも画面にも出さない * - HTML から { mode, system, messages, token } を POST で受け取り、Gemini に中継する * - 一時的な混雑(503/429/500)は自動リトライ。ダメなら予備モデルへ自動切替。 * * 【初期設定】スクリプト プロパティに * GEMINI_KEY = （Gemini APIキー） ← 必須 * ACCESS_TOKEN = （合言葉。HTML側と同じ文字列） ← 任意（未設定なら認証なし） * * 【重要】コードを書き換えたら必ず再デプロイ: * デプロイ → デプロイを管理 → 鉛筆 → バージョン「新しいバージョン」→ デプロイ * （※プロパティの変更は再デプロイ不要だが、コード変更は新バージョンが必要） */ var GEMINI_MODELS = ['gemini-3.5-flash', 'gemini-2.5-flash']; // 先頭から順に試す（予備込み） var MAX_RETRY = 3; function doGet() { return json_({ ok: true, models: GEMINI_MODELS }); } function doPost(e) { try { var props = PropertiesService.getScriptProperties(); var key = props.getProperty('GEMINI_KEY'); if (!key) throw new Error('GEMINI_KEY が未設定です'); var req = JSON.parse(e.postData.contents); // { mode, system, messages, token } var expected = props.getProperty('ACCESS_TOKEN'); if (expected && req.token !== expected) { return json_({ error: 'unauthorized' }); } var payload = { systemInstruction: { parts: [{ text: req.system || '' }] }, contents: (req.messages || []).map(function (m) { return { role: (m.role === 'assistant' ? 'model' : 'user'), parts: [{ text: String(m.content || '') }] }; }) }; if (req.mode === 'finalize') { payload.generationConfig = { responseMimeType: 'application/json', temperature: 0.2 }; } var data = callGemini_(key, payload); var text = ''; if (data.candidates && data.candidates[0] && data.candidates[0].content && data.candidates[0].content.parts) { text = data.candidates[0].content.parts.map(function (p) { return p.text || ''; }).join(''); } return json_({ text: text }); } catch (err) { return json_({ error: String(err && err.message ? err.message : err) }); } } /** モデルを順に・各モデルで数回リトライして呼ぶ */ function callGemini_(key, payload) { var lastErr = 'gemini error'; for (var mi = 0; mi < GEMINI_MODELS.length; mi++) { var url = 'https://generativelanguage.googleapis.com/v1beta/models/' + GEMINI_MODELS[mi] + ':generateContent?key=' + encodeURIComponent(key); for (var attempt = 0; attempt < MAX_RETRY; attempt++) { var res = UrlFetchApp.fetch(url, { method: 'post', contentType: 'application/json', payload: JSON.stringify(payload), muteHttpExceptions: true }); var code = res.getResponseCode(); var body = res.getContentText(); if (code === 200) return JSON.parse(body); lastErr = 'HTTP ' + code + ': ' + body; // 一時的な混雑系はリトライ（指数的に待つ） if (code === 503 || code === 429 || code === 500) { Utilities.sleep(800 * (attempt + 1)); // 0.8s → 1.6s → 2.4s continue; } // 400 など恒久エラーはリトライ無駄 → 次のモデルへ break; } } throw new Error(lastErr); } function json_(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
